% FUNCIONES
\pgfmathdeclarefunction{lorentz}{2}{%
  \pgfmathparse{(2/(#2*pi))/(1 + 4*(#1/#2)^2)}%
}
\pgfmathdeclarefunction{doppler}{2}{%
  \pgfmathparse{(2/#2)*(ln(2)/pi)^0.5*exp(-4*ln(2)*(#1/#2)^2)}%
}
